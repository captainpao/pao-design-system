## Usage Examples

```html
<!-- Basic Button Group -->
<pao-button-group
  .buttons=${[
    { label: 'Option 1' },
    { label: 'Option 2' },
    { label: 'Option 3' }
  ]}
  .defaultActive=${[0]}
></pao-button-group>

<!-- Multiple Selection -->
<pao-button-group
  .buttons=${[
    { label: 'Bold' },
    { label: 'Italic' },
    { label: 'Underline' }
  ]}
  selectionType="multiple"
  .defaultActive=${[0, 2]}
></pao-button-group>

<!-- Different Styles -->
<pao-button-group
  .buttons=${[
    { label: 'Option 1' },
    { label: 'Option 2' },
    { label: 'Option 3' }
  ]}
  styleType="solid"
  type="secondary"
  roundedCorners
></pao-button-group>

<!-- Event Handling -->
<pao-button-group
  .buttons=${[
    { label: 'Save' },
    { label: 'Cancel' }
  ]}
  @pao-selection-change=${(e) => console.log('Selection changed:', e.detail)}
></pao-button-group>
```

```javascript
// Controlled Component
const [activeButtons, setActiveButtons] = useState([0]);

const handleSelectionChange = (event) => {
  setActiveButtons(event.detail.active);
};

// In your template
html`
  <pao-button-group
    .buttons=${[
      { label: 'Option 1' },
      { label: 'Option 2' },
      { label: 'Option 3' }
    ]}
    .active=${activeButtons}
    @pao-active-change=${handleSelectionChange}
  ></pao-button-group>
`;
```
